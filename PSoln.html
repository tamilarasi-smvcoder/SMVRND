5. Power of Numbers
                        
public class Main 
{
    public static void main(String[] args) 
    {
        int N = 12 , R = 21;
        Solution ob = new Solution();
        System.out.println(ob.power(N,R));
    }
}
class Solution
{
    static long power(int N, int R)
    {
        if(R==0) return 1;
        if(R==1) return N;
        if(R%2==0)
        {
            long halfpow = power(N,R/2);
            return (long)(halfpow*halfpow)%1000000007;
        }
        else
        {
            return (long)(N*power(N,R-1))%1000000007;
        }
    }
}

6. Find GCD Using Euclide and Grabriel lame Algorithm
                        
public class Main 
{
    public static void main(String[] args) 
    {
        int A = 12 , B = 21;
        Solution ob = new Solution();
        System.out.println(ob.EcluidGCD(A,B));
    }
}
class Solution
{
    static long EuclideGCD(int A, int B)
    {
        while(A!=0&&B!=0)
        {
            if(A>B)
            A = A%B;
            else
            B = B%A;
        }
        if(A!=0)
        return A;
        else
        return B;
    }
}
